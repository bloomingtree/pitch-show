## 2. 数据格式设计

### 2.1 原生格式 (.pitch)

采用 JSON 格式，易于解析和扩展：

```typescript
interface PitchFile {
  // 文件元信息
  version: string;           // 格式版本号，如 "1.0"
  app: string;               // 应用标识，如 "pitch-show"
  createdAt: string;         // ISO 8601 时间戳
  checksum: string;          // SHA-256 校验和（不含此字段）

  // 歌曲信息
  song: {
    name: string;            // 歌曲显示名称
    originalFileName?: string; // 原始音频文件名
    duration: number;        // 总时长（秒）
    analyzedAt: string;      // 分析时间
  };

  // 音符数据
  notes: Note[];

  // 节拍数据（可选）
  beatData?: BeatData;

  // 用户设置（可选）
  settings?: {
    filterConfidence?: number;
    mergeGap?: number;
    colorScheme?: string;
  };
}

interface Note {
  pitchMidi: number;         // MIDI 音高 (0-127)
  startTimeSeconds: number;  // 开始时间（秒）
  durationSeconds: number;   // 持续时间（秒）
  amplitude: number;         // 振幅 (0-1)
  confidence?: number;       // 置信度 (0-1)
  pitchBends?: PitchBend[];  // 弯音数据
  isDynamic?: boolean;       // 是否动态音符
}

interface PitchBend {
  time: number;              // 相对音符开始的时间（秒）
  pitchBend: number;         // 弯音值（半音）
}

interface BeatData {
  bpm: number;               // 每分钟节拍数
  offset: number;            // 首拍偏移（秒）
  beats: Beat[];
}

interface Beat {
  time: number;              // 时间点（秒）
  isDownbeat: boolean;       // 是否强拍
}
```

## 3. 核心模块设计

### 3.1 SongExporter 类

```javascript
// src/js/songExporter.js

import { pack } from 'fflate';
import { PitchFormatHandler } from './formatHandlers/pitchFormat';
import { MidiFormatHandler } from './formatHandlers/midiFormat';

export class SongExporter {
  constructor() {
    this.handlers = {
      pitch: new PitchFormatHandler(),
      midi: new MidiFormatHandler(),
    };
  }

  /**
   * 导出单首歌曲
   * @param {Object} songData - 歌曲数据
   * @param {string} format - 导出格式 ('pitch' | 'midi')
   * @returns {Promise<Blob>} 导出的文件 Blob
   */
  async exportSingle(songData, format = 'pitch') {
    const handler = this.handlers[format];
    if (!handler) {
      throw new Error(`Unsupported format: ${format}`);
    }

    // 1. 获取处理器
    const content = await handler.serialize(songData);

    // 2. 创建 Blob
    const blob = new Blob([content], {
      type: handler.mimeType
    });

    return blob;
  }

  /**
   * 批量导出歌曲
   * @param {Array<Object>} songs - 歌曲数据数组
   * @param {string} format - 导出格式
   * @param {Function} onProgress - 进度回调
   * @returns {Promise<Blob>} ZIP 文件 Blob
   */
  async exportBatch(songs, format = 'pitch', onProgress) {
    const handler = this.handlers[format];
    const files = {};

    for (let i = 0; i < songs.length; i++) {
      const song = songs[i];
      const content = await handler.serialize(song);
      const fileName = this.sanitizeFileName(song.name) + handler.extension;
      files[fileName] = new Uint8Array(content);

      onProgress?.({
        current: i + 1,
        total: songs.length,
        currentSong: song.name
      });
    }

    // 压缩为 ZIP
    return new Promise((resolve, reject) => {
      pack(files, (err, data) => {
        if (err) reject(err);
        else resolve(new Blob([data], { type: 'application/zip' }));
      });
    });
  }

  /**
   * 触发文件下载
   * @param {Blob} blob - 文件 Blob
   * @param {string} fileName - 文件名
   */
  download(blob, fileName) {
    const url = URL.createObjectURL(blob);
    const link = document.createElement('a');
    link.href = url;
    link.download = fileName;
    link.click();
    URL.revokeObjectURL(url);
  }

  /**
   * 清理文件名（移除非法字符）
   */
  sanitizeFileName(name) {
    return name.replace(/[<>:"/\\|?*]/g, '_');
  }
}
```

### 3.2 SongImporter 类

```javascript
// src/js/songImporter.js

import { unpack } from 'fflate';
import { PitchFormatHandler } from './formatHandlers/pitchFormat';
import { MidiFormatHandler } from './formatHandlers/midiFormat';
import { SongValidator } from './validators/songValidator';
import SongStore from '@/store/Song';

export class SongImporter {
  constructor() {
    this.handlers = {
      pitch: new PitchFormatHandler(),
      midi: new MidiFormatHandler(),
    };
    this.validator = new SongValidator();
  }

  /**
   * 导入单个文件
   * @param {File} file - 文件对象
   * @returns {Promise<Object>} 解析后的歌曲数据
   */
  async importFile(file) {
    // 1. 检测文件类型
    const format = this.detectFormat(file);
    if (!format) {
      throw new ImportError('UNSUPPORTED_FORMAT', 'Unsupported file format');
    }

    // 2. 读取文件内容
    const content = await this.readFile(file);

    // 3. 解析内容
    const handler = this.handlers[format];
    const songData = await handler.deserialize(content);

    // 4. 校验数据
    const validation = this.validator.validate(songData);
    if (!validation.valid) {
      throw new ImportError('INVALID_DATA', validation.errors.join(', '));
    }

    return songData;
  }

  /**
   * 批量导入（支持 ZIP）
   * @param {File} file - ZIP 文件或单文件
   * @param {Function} onProgress - 进度回调
   * @returns {Promise<Array<Object>>} 解析后的歌曲数据数组
   */
  async importBatch(file, onProgress) {
    if (file.name.endsWith('.zip')) {
      return this.importFromZip(file, onProgress);
    } else {
      const songData = await this.importFile(file);
      return [songData];
    }
  }

  /**
   * 从 ZIP 导入
   */
  async importFromZip(file, onProgress) {
    const buffer = await file.arrayBuffer();
    const files = await new Promise((resolve, reject) => {
      unpack(new Uint8Array(buffer), (err, files) => {
        if (err) reject(err);
        else resolve(files);
      });
    });

    const songs = [];
    const fileNames = Object.keys(files);

    for (let i = 0; i < fileNames.length; i++) {
      const fileName = fileNames[i];
      if (!fileName.endsWith('.pitch') && !fileName.endsWith('.mid')) {
        continue;
      }

      try {
        const content = files[fileName].buffer;
        const format = fileName.endsWith('.pitch') ? 'pitch' : 'midi';
        const handler = this.handlers[format];
        const songData = await handler.deserialize(content);

        const validation = this.validator.validate(songData);
        if (validation.valid) {
          songs.push(songData);
        }

        onProgress?.({
          current: i + 1,
          total: fileNames.length,
          currentFile: fileName
        });
      } catch (err) {
        console.warn(`Failed to import ${fileName}:`, err);
      }
    }

    return songs;
  }

  /**
   * 保存导入的歌曲
   * @param {Object} songData - 歌曲数据
   * @param {Object} options - 保存选项
   * @returns {Promise<string>} 保存后的歌曲名
   */
  async saveImportedSong(songData, options = {}) {
    const { onConflict = 'rename' } = options;

    // 检查名称冲突
    const existing = await SongStore.get(songData.song.name);
    let finalName = songData.song.name;

    if (existing) {
      switch (onConflict) {
        case 'skip':
          return null;
        case 'overwrite':
          await SongStore.remove(songData.song.name);
          break;
        case 'rename':
        default:
          finalName = await this.generateUniqueName(songData.song.name);
          break;
      }
    }

    // 保存到 IndexedDB
    await SongStore.add(
      finalName,
      songData.song.originalFileName || finalName,
      songData.notes,
      songData.beatData
    );

    return finalName;
  }

  /**
   * 检测文件格式
   */
  detectFormat(file) {
    const ext = file.name.split('.').pop().toLowerCase();
    if (ext === 'pitch') return 'pitch';
    if (ext === 'mid' || ext === 'midi') return 'midi';
    return null;
  }

  /**
   * 读取文件内容
   */
  async readFile(file) {
    return new Promise((resolve, reject) => {
      const reader = new FileReader();
      reader.onload = () => resolve(reader.result);
      reader.onerror = () => reject(reader.error);

      if (file.name.endsWith('.pitch')) {
        reader.readAsText(file);
      } else {
        reader.readAsArrayBuffer(file);
      }
    });
  }

  /**
   * 生成唯一名称
   */
  async generateUniqueName(baseName) {
    const allSongs = await SongStore.getAll();
    const regex = new RegExp(`^${baseName}(?: \\((\\d+)\\))?$`);
    let maxNum = 0;

    for (const song of allSongs) {
      const match = song.name.match(regex);
      if (match) {
        const num = match[1] ? parseInt(match[1], 10) : 0;
        maxNum = Math.max(maxNum, num);
      }
    }

    return `${baseName} (${maxNum + 1})`;
  }
}

/**
 * 自定义导入错误
 */
class ImportError extends Error {
  constructor(code, message) {
    super(message);
    this.code = code;
    this.name = 'ImportError';
  }
}
```

### 3.3 格式处理器

#### PitchFormatHandler

```javascript
// src/js/formatHandlers/pitchFormat.js

const FORMAT_VERSION = '1.0';

export class PitchFormatHandler {
  extension = '.pitch';
  mimeType = 'application/json';

  /**
   * 序列化歌曲数据
   */
  async serialize(songData) {
    const pitchFile = {
      version: FORMAT_VERSION,
      app: 'pitch-show',
      createdAt: new Date().toISOString(),
      song: {
        name: songData.name,
        originalFileName: songData.originalFileName,
        duration: this.calculateDuration(songData.notes),
        analyzedAt: songData.date || new Date().toISOString(),
      },
      notes: songData.notes,
      beatData: songData.beatData || null,
    };

    // 计算校验和
    const content = JSON.stringify(pitchFile);
    pitchFile.checksum = await this.calculateChecksum(content);

    return JSON.stringify(pitchFile, null, 2);
  }

  /**
   * 反序列化歌曲数据
   */
  async deserialize(content) {
    let data;

    if (typeof content === 'string') {
      data = JSON.parse(content);
    } else if (content instanceof ArrayBuffer) {
      const text = new TextDecoder().decode(content);
      data = JSON.parse(text);
    }

    // 版本检查
    if (!this.isVersionCompatible(data.version)) {
      throw new Error(`Incompatible version: ${data.version}`);
    }

    // 校验和验证（可选）
    if (data.checksum) {
      const expected = data.checksum;
      delete data.checksum;
      const actual = await this.calculateChecksum(JSON.stringify(data));
      if (expected !== actual) {
        console.warn('Checksum mismatch, file may be corrupted');
      }
    }

    return data;
  }

  /**
   * 计算校验和
   */
  async calculateChecksum(content) {
    const encoder = new TextEncoder();
    const data = encoder.encode(content);
    const hashBuffer = await crypto.subtle.digest('SHA-256', data);
    const hashArray = Array.from(new Uint8Array(hashBuffer));
    return hashArray.map(b => b.toString(16).padStart(2, '0')).join('');
  }

  /**
   * 版本兼容性检查
   */
  isVersionCompatible(version) {
    if (!version) return true; // 早期版本无版本号
    const [major] = version.split('.');
    const [currentMajor] = FORMAT_VERSION.split('.');
    return major === currentMajor;
  }

  /**
   * 计算总时长
   */
  calculateDuration(notes) {
    if (!notes || notes.length === 0) return 0;
    return Math.max(...notes.map(n => n.startTimeSeconds + n.durationSeconds));
  }
}
```

#### MidiFormatHandler

```javascript
// src/js/formatHandlers/midiFormat.js

// 注意：需要安装 midi-writer-js
// npm install midi-writer-js

import MidiWriter from 'midi-writer-js';

export class MidiFormatHandler {
  extension = '.mid';
  mimeType = 'audio/midi';

  /**
   * 序列化歌曲数据为 MIDI
   */
  async serialize(songData) {
    const track = new MidiWriter.Track();

    // 设置速度
    const bpm = songData.beatData?.bpm || 120;
    track.setTempo(bpm);

    // 转换音符
    const ticksPerSecond = (bpm * 480) / 60; // 480 = 标准 PPQN

    for (const note of songData.notes) {
      const midiNote = new MidiWriter.NoteEvent({
        pitch: note.pitchMidi,
        duration: this.secondsToTicks(note.durationSeconds, ticksPerSecond),
        startTick: this.secondsToTicks(note.startTimeSeconds, ticksPerSecond),
        velocity: this.amplitudeToVelocity(note.amplitude),
      });
      track.addEvent(midiNote);
    }

    const write = new MidiWriter.Writer(track);
    return write.buildFile();
  }

  /**
   * 从 MIDI 反序列化（基础实现）
   */
  async deserialize(content) {
    // MIDI 导入相对复杂，P1 阶段可仅支持导出
    throw new Error('MIDI import not supported in this version');
  }

  /**
   * 秒转换为 MIDI ticks
   */
  secondsToTicks(seconds, ticksPerSecond) {
    return Math.round(seconds * ticksPerSecond);
  }

  /**
   * 振幅转换为 MIDI 力度
   */
  amplitudeToVelocity(amplitude) {
    // amplitude: 0-1, velocity: 1-127
    return Math.round(amplitude * 126) + 1;
  }
}
```

```javascript
// 使用流式处理大文件
async function* streamExport(songs) {
  for (const song of songs) {
    yield {
      name: song.name,
      data: await exporter.exportSingle(song)
    };
  }
}

// 分块压缩
async function exportInChunks(songs, chunkSize = 10) {
  const chunks = [];
  for (let i = 0; i < songs.length; i += chunkSize) {
    chunks.push(songs.slice(i, i + chunkSize));
  }

  const results = [];
  for (const chunk of chunks) {
    const blob = await exporter.exportBatch(chunk);
    results.push(blob);
  }

  return results;
}
```

```javascript
function downloadAndCleanup(blob, filename) {
  const url = URL.createObjectURL(blob);
  const link = document.createElement('a');
  link.href = url;
  link.download = filename;
  link.click();

  // 延迟释放，确保下载开始
  setTimeout(() => URL.revokeObjectURL(url), 1000);
}
```

### 6.3 缓存策略

```javascript
// 导出结果缓存（短时间内重复导出）
const exportCache = new Map();

function getCachedExport(songName, format) {
  const key = `${songName}:${format}`;
  const cached = exportCache.get(key);

  if (cached && Date.now() - cached.timestamp < 60000) {
    return cached.blob;
  }

  return null;
}

function setCachedExport(songName, format, blob) {
  const key = `${songName}:${format}`;
  exportCache.set(key, {
    blob,
    timestamp: Date.now()
  });

  // 限制缓存大小
  if (exportCache.size > 50) {
    const oldest = exportCache.keys().next().value;
    exportCache.delete(oldest);
  }
}
```
